SQL Tutorials

Learn SQL using real retail data — click any query to load it in the explorer

Beginner

Basic SELECT

Retrieve all records from the products table.

SELECT * FROM products;
Beginner

Filtering with WHERE

Find all Electronics products under $100.

SELECT product_name, price
FROM products
WHERE category = 'Electronics'
  AND price < 100;
Beginner

ORDER BY & LIMIT

Get the top 5 most expensive products.

SELECT product_name, category, price
FROM products
ORDER BY price DESC
LIMIT 5;
Intermediate

Aggregate Functions

Count products and calculate average price per category.

SELECT category,
  COUNT(*) AS total_products,
  ROUND(AVG(price), 2) AS avg_price,
  MAX(price) AS max_price
FROM products
GROUP BY category
ORDER BY avg_price DESC;
Intermediate

JOIN Tables

List all transactions with customer and product names.

SELECT t.transaction_id,
  c.first_name || ' ' || c.last_name AS customer,
  p.product_name,
  t.quantity,
  p.price * t.quantity AS total_amount,
  t.transaction_date
FROM transactions t
JOIN customers c ON t.customer_id = c.customer_id
JOIN products p ON t.product_id = p.product_id
ORDER BY t.transaction_date DESC;
Intermediate

Revenue by Store

Calculate total revenue for each store location.

SELECT s.location,
  s.manager_name,
  COUNT(t.transaction_id) AS total_sales,
  ROUND(SUM(t.quantity * p.price), 2) AS revenue
FROM transactions t
JOIN stores s ON t.store_id = s.store_id
JOIN products p ON t.product_id = p.product_id
GROUP BY s.store_id
ORDER BY revenue DESC;
Advanced

Monthly Trends

Aggregate monthly sales revenue using strftime.

SELECT
  strftime('%Y-%m', transaction_date) AS month,
  COUNT(*) AS transactions,
  ROUND(SUM(t.quantity * p.price), 2) AS revenue
FROM transactions t
JOIN products p ON t.product_id = p.product_id
GROUP BY month
ORDER BY month;
Advanced

Subquery — Above Avg Spenders

Find customers who spent more than the average transaction amount.

SELECT c.first_name || ' ' || c.last_name AS customer,
  ROUND(SUM(t.quantity * p.price), 2) AS total_spent
FROM transactions t
JOIN customers c ON t.customer_id = c.customer_id
JOIN products p ON t.product_id = p.product_id
GROUP BY t.customer_id
HAVING total_spent > (
  SELECT AVG(sub.quantity * sp.price)
  FROM transactions sub
  JOIN products sp ON sub.product_id = sp.product_id
)
ORDER BY total_spent DESC;
Advanced

Window Function — Running Total

Calculate a running total of revenue ordered by date.

SELECT transaction_date,
  ROUND(t.quantity * p.price, 2) AS sale_amount,
  ROUND(SUM(t.quantity * p.price)
    OVER (ORDER BY transaction_date), 2) AS running_total
FROM transactions t
JOIN products p ON t.product_id = p.product_id
ORDER BY transaction_date;
